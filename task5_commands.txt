mkdir -p Task5/.github/workflows
mkdir -p .github/workflows
cat > Task5/.github/workflows/deploy.yml << 'EOF'
name: Deploy Node.js API

on:
  push:
    branches:
      - main

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository code
        uses: actions/checkout@v4

      - name: Set up Node.js 18
        uses: actions/setup-node@v4
        with:
          node-version: "18"

      - name: Install project dependencies
        run: |
          cd Task4
          npm install

      - name: Build Docker image
        run: |
          cd Task4
          docker build -t nodejs-api .

      - name: Deploy application
        run: echo "Deploying application to server..."
EOF
cp Task5/.github/workflows/deploy.yml .github/workflows/deploy.yml
git add .
git commit -m "This is the new CI/CD pipeline for CA2"
git push -u origin master
